# Data Structures

!! Region
  name:      string
  has_many   :zones

Zone
  name:      string
  belongs_to :region
  has_many   :bgms, as: :listenable
  has_many   :subzones, class_name: "Zone", foreign_key: "mainzone_id"
  belongs_to :mainzone, class_name: "Zone", optional: true

!! Duty_Type
  name:      string
  has_many   :duties

Duty
  name:      string
  level:     integer
  belongs_to :duty-type
  belongs_to :expansion
  has_many   :bgms, as: :instance

Expansion
  number:    integer
  name:      string
  has_many   :duties

!! Track
  name:      string
  sample:    string
  has_many   :bgms
  has_one    :orchestrion

Bgm
  belongs_to :track
  context:   string
  belongs_to :instance, polymorphic: true

  add_index  :bgm, [:instance_type, :instance_id]

!! Orchestrion_Type
  name:      string
  has_many   :orchestrions

Orchestrion
 # name:      string
  belongs_to :track
  belongs_to :orchestrion-type
